\section{asyncIO}

\begin{frame}{Motivation}
        \tb{IO-Operationen} sind standardmmäßig \tb{synchron} und damit blockierend.
		\newline
		Simple Beispiele: Lesen aus einer Datei, \pp{time.sleep()}
		
		\medskip
		
		Während dieser \tb{Wartezeit} könnte an einer anderen Stelle doch bereits \tb{sinnvolle} Arbeit verrichtet werden.
		
		\medskip
		
		Dies ist insbesondere bei Programmen wünschenswert, welche auf \tb{Anfragen} oder Events im Allgemeinen  \tb{reagieren} sollen. 

	
\end{frame}

\begin{frame}{Was ist asyncIO?}
        \pp{asyncio} ist eine \tb{built-in} Library von Python.
		\newline
		Mit ihr kommen zwei neuen \tb{Keywords}: \pp{async} \& \pp{await}.
		
		\medskip
		
		Sie bietet einen \tb{asynchronen} Programm-Loop, welcher in \tb{einem Thread} läuft.
		
		\medskip
		
		Durch das Verwenden von \pp{asyncio} kann die aktuell bearbeitete Funktion an bestimmten Stellen \tb{pausiert} werden, um in einer anderen Funktion weiterzuarbeiten, welche nicht auf IO wartet.

	
\end{frame}


\begin{frame}{Was macht async jetzt?}
        Funktionen, die asynchron ausgeführt werden sollen, werden durch \pp{async def(...):} \tb{gekennzeichnet}.
		\newline
		Innerhalb dieser Funktionen dürfen andere asynchrone Funktionen aufgerufen und auch erwartet werden.
		
		\medskip
		
		Wird eine \pp{async} Funktion \tb{aufgerufen}, gibt sie ein \tb{Coroutine}-Objekt zurück, statt die eigentliche Funktion auszuführen. 
		
		\medskip
		
		Das Keyword \pp{await} \tb{blockiert} den Programmfluss der Funktion so lange, bis das Ergebnis des \pp{awaitable}\footnote{Async-Tasks sind auch awitable: \url{https://docs.python.org/3/library/asyncio-task.html\#awaitables}} vorliegt. Dieses wird dann zurückgegeben.
		
		\medskip
		
		Spannend sind die async-\tb{Tasks}, wodurch mehrere Coroutinen \tb{quasi nebenläufig}\newline
		laufen können. 


\end{frame}


\begin{frame}{Awaitable}

		Ein \pp{awaitable}\furl{https://docs.python.org/3/library/asyncio-task.html\#awaitables} definiert sich dadurch, dass es \pp{\_\_await\_\_()}\footnote{Zudem existieren \_\_aiter\_\_(), \_\_aenter\_\_() und weitere} implementiert. \newline
		Dadurch kann es mit \pp{await} erwartet werden.
		
		\medskip
		
		Die drei built-in Typen von \pp{awaitables} sind \tb{Coroutine}, \tb{Task} und \tb{Future}\footnote{Wird hier nicht weiter besprochen.}.

\end{frame}


\begin{frame}[fragile]{Async-Loop beginnen}
        Der asynchrone Programmfluss wird durch \tb{Aufrufen} von \pp{asyncio.run()} gestartet.

		\medskip
		
		Die Funktion nimmt eine \tb{Coroutine} entgegen.
		\begin{lstlisting}[language=pytiny]
asyncio.run(main())
		\end{lstlisting}
		$\rightarrow$ Die zu \tb{übergebene} Funktion wird unintuitiver Weise \tb{\glqq ausgeführt\grqq} übergeben.
		
		\medskip
		
		Der \tb{Start} des Loops ist \tb{blockierend}.

\end{frame}


\begin{frame}{Async-Coroutine}
        Ein \tb{Coroutine}-Objekt\furl{https://docs.python.org/3/library/asyncio-task.html\#coroutines} wird durch den \tb{\glqq Aufruf\grqq~} einer \pp{async}-Funktion \tb{erzeugt}.

		\medskip
		
		Die eigentliche Funktion wird \tb{erst ausgeführt}, wenn es an einen \pp{Task} \tb{übergeben} oder mit \pp{await} \tb{erwartet} wird.
		
		\medskip
		
		Ein \pp{coroutine} muss vor Ende der erstellenden Funktion erwartet werden.
		
		\medskip
		
		Die Rückgabe einer nicht erwarteten \pp{coroutine} ist nicht möglich.
\end{frame}

\begin{frame}{Async-Tasks}
        Eine async-\tb{Task}\furl{https://docs.python.org/3/library/asyncio-task.html\#task-object} wird durch \pp{asyncio.create_task()} zur Bearbeitung gescheduled.
		\newline
		Hierdurch kann \tb{nebenläufiges Verhalten} erzielt werden.
		
		\medskip
		
		Sein Ende und das \tb{Ergebnis} können mit \pp{await} \tb{abgewartet} werden.
		\newline 
		Hierdruch kann sein \tb{Bearbeitungszustand} mit \pp{.done()} Abgefragt werden.
		\newline
		Das Ergebnis eines fertigen Tasks kann mit \pp{.result()} \tb{abgeholt} werden.
		
		\medskip
		
		\pp{asyncio.gather()} erwartet das Ende aller übergeben Tasks und gibt die Ergebnisse zurück.
		\newline
		Die Reihenfolge der Rückgabe entspricht der Reihenfolge der Übergabe.
		
		\medskip
		
		Ein Task muss theoretisch niemals awaited werden.
		

\end{frame}

\begin{frame}[fragile]{Async-Tasks}
	\begin{lstlisting}[language=pytiny]
import asyncio

async def sleep(name: str, st=1):
    """ helper to execute an 'IO-heavy' task (sleeping) """
    print(f"[{name}] Before await")
    await asyncio.sleep(st)
    print(f"[{name}] After await")
    return name

async def main():
    task1 = asyncio.create_task(sleep("call1"))  # create task
    task2 = asyncio.create_task(sleep("call2"))
    print(type(task1))  # >> <class '_asyncio.Task'>

    res1, res2 = await asyncio.gather(task1, task2)  # wait for tasks to finish
    print(res1, res2)  # >> call1, call2

# start main async-loop
asyncio.run(main())
	\end{lstlisting}

\end{frame}


\begin{frame}[fragile]{Async-Taskgroup}
        Das selbe Beispiel wie oben mit Hilfe von \tb{TaskGroup}\furl{https://docs.python.org/3/library/asyncio-task.html\#task-groups}.

		\medskip
		
		In dieser können \tb{mehrere} Tasks einfach gescheduled werden.
		
		\medskip
		
		Beim \tb{Verlassen} des Context-Blocks wird sicher gestellt, dass alle \tb{Tasks fertig} sind.

	\begin{lstlisting}[language=pytiny]
async def main():
    print(f"started at {time.strftime('%X')}")  # %X -> hh:mm:ss
    async with asyncio.TaskGroup() as tg:
        task1 = tg.create_task(sleep('call1'))

        task2 = tg.create_task(sleep("call2"))

    print(f"ended at {time.strftime('%X')}")

    print(task1.result())
    
asyncio.run(main())
	\end{lstlisting}

\end{frame}


\begin{frame}{Wo können Async-Funktion untebrochen werden?}

		Kurz: Überall, wo eine \tb{Interaktion} mit \pp{asyncio} passiert.
		
		\medskip
		
		 Der Programmfluss kann bei Aufrufen von \pp{async}-Funktionen und dem Nutzen von \pp{await} angehalten und woanders fortgeführt werden.

\end{frame}


\begin{frame}{Schützen von Ressourcen}
        Wenn mehrere Funktionen \tb{\glqq zeitgleich\grqq}~laufen können, müssen geteilte \tb{Ressourcen} gegebenenfalls \tb{geschützt werden}\footnote{\url{https://docs.python.org/3/library/asyncio-sync.html\#synchronization-primitives}}.

		\medskip
		
		Anders als bei Multithreading müssen nicht alle Zugriffe geschützt werden.
		
		\medskip
		
		Es sollte drauf geachtet werden, eine Ressource zu \tb{schützen}, wenn ihre Konsistenz über mehrere \tb{async-Aufrufe hinweg} notwendig ist.
		
		\medskip
		
		Zudem muss beachtet werden, dass Libraries potentiell \glqq versteckt\grqq~\pp{asyncio} verwenden, wodurch die Kontrolle an die Library zurückfällt.

\end{frame}


\begin{frame}[fragile]{Timeout}
    	Mit \pp{asyncio.timeout()}\footnote{\url{https://docs.python.org/3/library/asyncio-task.html\#timeouts}} kann festgelegt werden, wann ein \pp{Task} beendet werden soll.

		\medskip
		
		Mit \pp{asyncio.Timeout} kann ein Timeout festgelegt und rescheduled werden.
		\newline
		Zudem kann abgefragt werden, ob das Timeout erreicht wurde.

    \begin{lstlisting}[language=pytiny]
async def main():
    try:
        async with asyncio.timeout(10):
            await long_running_task()
    except TimeoutError:
        print("The long operation timed out, but we've handled it.")
    \end{lstlisting}

\end{frame}

\section{Discord Bots}
\subsection{Async in einer Folie}

\begin{frame}{Async: Informell in vier Punkten + Fazit}
	\begin{itemize}
		\item \tb{Asynchrone} Programme können zwischen der \tb{Ausführung} verschiedener Funktionen \tb{wechseln}, wenn eine Funktion \tb{blockierend} auf (i.d.R.) I/O wartet.
		
		\pause
		
		\item[] Hierbei läuft das gesamte Programm immernoch in \tb{einem} Thread\footnote{Dennoch ist die Ausführung des Programms unvorhersehbar, daher kann es notwendig sein, Ressourcen zu schützen!}.
		
		\pause
		
		\item Ist eine \tb{Funktion} als \pp{async} gekennzeichnet, bedeutet dies, dass sie irgendwo auf (i.d.R.) I/O \tb{warten} wird.
		
		\pause
		
		
		\item Solche Methoden nennen sich \tb{Coroutinen}.
		
		\pause
		
		\item Coroutinen \tb{müssen}, im Kontext von Discord.py, direkt mit \pp{await} \tb{abgewartet} werden.
		
		\pause
		
		\item[] \tb{Sobald} das Programm auf \pp{await} trifft, kann es an einer \tb{anderen} Stelle weiter arbeiten und zu diesem Aufruf \tb{zurückkehren}, sobald die Coroutine abgeschlossen ist.
		
		\item \tb{Fazit}: Durch \pp{asyncIO} wird \tb{keine Zeit} mit Warten auf I/O \tb{verschwendet}.
		
	\end{itemize}
\end{frame}

\subsection{Event-basierte Operationen}
\begin{frame}{Event-basierte Operationen: Überblick}
	    \tb{Event}-basierte Operationen sind Operationen, die der Bot durchführt, wenn ein\newline
	     bestimmtes Event \tb{aufgetreten} ist.
	
		\medskip
		\pause
		
		Bots können auf eine ganze Reihe von Events\furl{https://discordpy.readthedocs.io/en/latest/api.html\#event-reference} hören. Unter Anderem:
		%
		\begin{itemize}
			\item Empfang einer Nachricht \pp{on_message}
			\item Join eines Members \pp{on_member_join}
			\item Hinzufügen / Entfernen einer Reaction \pp{on_reaction_add} bzw. \pp{_remove}.
			\item Die Veränderung eines Members \pp{on_member_update}
			\item Oder auch einfach das Tippen in einem Channel \pp{on_typing}
		\end{itemize}
		
		\medskip
		\pause
		
		Events werden durch das \tb{Überschreiben} der Methode des Bots gehandelt.
		
		\medskip
		\pause
		
		Hinweis: Viele dieser Events erfordern \tb{Privileged} Gateway Intents\furl{https://discord.com/developers/docs/topics/gateway\#privileged-intents}\furl{https://support-dev.discord.com/hc/en-us/articles/6207308062871-What-are-Privileged-Intents-}.

\end{frame}

bot = MyBot()
def start_bot():
    bot.run(os.getenv("TOKEN"))
	\end{lstlisting}

\begin{frame}{Weitere Kategorie von Events: Klassische Commands}
	    Eine spezifischere Teilkategorie von Events sind \tb{Commands}\footnote{Ein Beispiel eines Commands findet sich auf dem Beispiel zu \texttt{Cogs}, bei Commands außerhalb von Cogs muss \texttt{@bot.command()} als Decorator genutzt werden!}, mit welchen\newline
	     Nutzer:innen gezielt mit einem Bot interagieren.
	
		\medskip
		\pause
		
		Hierbei existiert der \tb{klassische} Message-Command\furl{https://discordpy.readthedocs.io/en/stable/ext/commands/api.html\#discord.ext.commands.command} mit der \tb{Syntax} \pp{prefix|command}.
		\newline
		Also simple \tb{Nachrichten}\footnote{
			Da die Commands auf Nachrichten basieren, muss beim Überschreiben der \texttt{on\_message()}-Funktion ein Aufruf von \texttt{process\_commands(message)} enthalten sein.
		}, welche einer besonderen Syntax folgen.
		
		\medskip
		\pause
		
		Die Interaktion ist ähnlich einer mit \tb{Command-Line} ohne Auto-Completion.

\begin{frame}{Application Commands}
	    Die neuere Art der Interaktion sind die \tb{Application-Commands}\furl{
		https://discordpy.readthedocs.io/en/latest/interactions/api.html?highlight=app_commands\#decorators
		}\footnote{
			Aufgrund des einheitlichen \texttt{/}-Prefixes oft einfach Slash-Commands genannt.
		}.
		
		\medskip
		\pause
		
		Sie sind direkt in Discord eingebaut und bieten \tb{Vorteile} gegenüber klassischen Commands:
		
		\medskip
		\pause
		
		\begin{itemize}
			\item Native \tb{Übersicht} aller verfügbaren Commands direkt in Discords UI.
			
			\medskip
			\pause
			
			\item \tb{Kommunikation} und \tb{Validierung} der verfügbaren Parameter inklusive Typ.
			
			\medskip
			\pause
			
			\item Ausführung und Antwort kann für andere Nutzer:innen \tb{unsichtbar} sein. 
			
			\medskip
			\pause
			
			\item In Discord eingebautes Rechte-System für die Nutzung von Commands.
		\end{itemize}
		
		\medskip
		\pause
		
		Aus der Developer-Community folgte einiges an \tb{Kritik}, da Slash-Commands die Einführung des \pp{Message Content Intent} ermöglichten und damit die allgemeine Freiheit der Bots einschränken. Discord begründet dies vor allem mit Datenschutz.

\end{frame}

\begin{frame}[fragile]{Rich - Progress Bars II}
	Das \tb{Progress}-Objekt ist eine \tb{Aggregation} aus mehreren Objekten\furl{https://rich.readthedocs.io/en/stable/progress.html\#columns}. 
	Es definiert das Aussehen der Ladebalken. Eine Auswahl:
	\pause
	\begin{itemize}
		\item \pp{BarColumn} die eigentliche Statusleiste
		\pause
		% TODO: besseres wort für rad
		\item \pp{SpinnerColumn} eine Art Rad, welches sich dauerhaft bewegt.
		\pause
		\item \pp{TimeRemainingColumn} automatische Schätzung der verbleibenden Zeit.
		\pause
	\end{itemize}
	
	\nll{Außerhalb des zusammengesetzten Objekts existieren können. \\
		Lebensdauer der Komponentenobjekte ist unabhängig.\\}
	
	\nsl{Eine "hat-eine"-Beziehung, bei der das zusammengesetzte Objekt auf bestehende Objekte verweist\\
		Ein Progress-Objekt enthält Verweise auf verschiedene Spaltenobjekte (TextColumn, SpinnerColumn, etc.), aber diese Spaltenobjekte können auch außerhalb des Progress-Objekts existieren und wiederverwendet werden.}
\end{frame}


\begin{frame}[fragile]{Rich - Progress Bars III}

	Auf ein Progress-Objekt können \tb{Tasks}%
	\footnote{https://rich.readthedocs.io/en/stable/reference/progress.html\#rich.progress.Task}
	mit \pp{.add_task()} \tb{registriert} werden.
	
	\medskip
	
	Als return gibt es eine \tb{TaskID}, diese wird zum \tb{Aktualisieren} des Fortschritts über das Progress-Objekt verwendet.
	\pause
	
	\medskip
	
	Die \tb{update}-Funktion bietet \pp{completed} für absolute Schritte und \pp{advance} für inkrementelle Schritte.\furl{https://rich.readthedocs.io/en/stable/reference/progress.html\#rich.progress.Progress.update}
	\pause
	
	\medskip
	
	Im Progress definierte \tb{Strings} haben lesenden \tb{Zugriff} auf Attribute der erstellten Tasks.
\pause
\end{frame}
