% this defines a /tikz/socket style that takes one optional argument,
%  i.e. the rotation of the socket (which, again, is zero, if the argument is omitted

    socket/.style={
        draw,
        thick,
        text width=0cm,
        minimum height=0cm,
        minimum size=0.8cm,
        rotsocket,
        socket rotate=#1
    },
    socket/.default=0
}

    plug/.style={
        draw,
        thick,
        text width=0cm,
        minimum height=0cm,
        minimum size=0.8cm,

    },
    plug/.default=0
}
